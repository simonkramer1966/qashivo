Score(i,c,t,k) =
α·P(pay|i,c,t,k) − β·F(friction|c,k) − γ·R(risk|i,c,k) + δ·U(urgency|portfolio)

// utils/scheduler.ts
import { addHours, isBefore } from "date-fns";
type Channel = "email" | "sms" | "call";
type Candidate = { t: Date; k: Channel; score: number; reason: string };

export function scheduleNextTouch(ctx: {
  tenant: { targetDSO: number; tz: string; quietHours: [number, number]; maxDailyTouches: number; urgencyFactor: number; };
  customer: { segment: string; priors: any; channelPrefs: Partial<Record<Channel, boolean>>; };
  invoice: { amount: number; dueAt: Date; issuedAt: Date; lastTouchAt?: Date; lastChannel?: Channel; ageDays: number; dispute: boolean; };
  history: { medLag: number; p75Lag: number; volatility: number; weekdayEffect: number[]; channel: Record<Channel, { replyRate: number; openRate?: number; }>; };
  constraints: { now: Date; minGapHours: number; allowedChannels: Channel[]; countryCode: string; };
}): Candidate {
  const cands: Candidate[] = [];
  const baseWindowHours = Math.max(24, Math.round(0.25 * (ctx.history.medLag || 14) * 24));
  const horizon = [12, 24, 36, 48, 72]; // hours from now to consider
  const channels: Channel[] = (ctx.constraints.allowedChannels || ["email","sms","call"])
    .filter(k => ctx.customer.channelPrefs[k] !== false);

  for (const h of horizon) {
    for (const k of channels) {
      const t = addHours(ctx.constraints.now, h);
      if (!passesQuietHours(t, ctx.tenant.quietHours, ctx.tenant.tz)) continue;
      if (!gapSinceLast(ctx.invoice.lastTouchAt, t, ctx.constraints.minGapHours)) continue;
      const pPay = estimatePPay(ctx, t, k);           // 0..1
      const friction = frictionCost(ctx, k);          // 0..1
      const risk = complianceRisk(ctx, t, k);         // 0..1
      const urgency = ctx.tenant.urgencyFactor || 0;  // 0..1+
      const α=1.0, β=0.35, γ=0.6, δ=0.4 + urgency;
      const score = α*pPay - β*friction - γ*risk + δ*urgencyBoost(ctx, t);
      cands.push({ t, k, score, reason: `p=${pPay.toFixed(2)} f=${friction.toFixed(2)} r=${risk.toFixed(2)} u=${urgency.toFixed(2)}` });
    }
  }
  cands.sort((a,b)=>b.score-a.score);
  return cands[0] || { t: addHours(ctx.constraints.now, 24), k: "email", score: 0, reason: "fallback" };
}

function estimatePPay(ctx:any, t:Date, k:Channel){
  const age = ctx.invoice.ageDays;
  const dueIn = Math.ceil((ctx.invoice.dueAt.getTime()-t.getTime())/(24*3600e3));
  const weekday = t.getDay();
  const base = sigmoid( 2.0 - 0.04*age + 0.03*ctx.history.weekdayEffect[weekday] );
  const chBoost = (ctx.history.channel?.[k]?.replyRate || 0.05) * 0.6;
  const amountPenalty = Math.min(0.25, Math.log10(Math.max(1, ctx.invoice.amount))/20);
  const seasonAdj = 0; // MVP
  const priorAdj = (ctx.customer.priors?.pPayBase ?? 0.02);
  return clamp(base + chBoost - amountPenalty + seasonAdj + priorAdj, 0, 1);
}
function frictionCost(ctx:any, k:Channel){
  const lowResp = 0.7 - (ctx.history.channel?.[k]?.replyRate || 0.05);
  const lastRepeat = ctx.invoice.lastChannel===k ? 0.1 : 0;
  return clamp(lowResp + lastRepeat, 0, 1);
}
function complianceRisk(ctx:any, t:Date, k:Channel){
  if (ctx.invoice.dispute) return 0.8;
  return isBefore(t, ctx.invoice.dueAt) && k==="call" ? 0.15 : 0.05;
}
function urgencyBoost(ctx:any, t:Date){
  // Sooner touches get slightly higher boost when portfolio DSO is above target
  const hours = Math.max(1, (t.getTime()-ctx.constraints.now.getTime())/3600e3);
  return clamp((ctx.tenant.urgencyFactor||0)* (1/Math.sqrt(hours)), 0, 1);
}
const clamp = (x:number,a:number,b:number)=>Math.max(a,Math.min(b,x));
const sigmoid = (z:number)=>1/(1+Math.exp(-z));
function passesQuietHours(t:Date, [qStart,qEnd]:[number,number], tz:string){
  const hour = t.getUTCHours(); // MVP: ignore tz offset or precompute
  if (qStart<qEnd) return !(hour>=qStart && hour<qEnd);
  return !(hour>=qStart || hour<qEnd);
}
function gapSinceLast(last?:Date, t?:Date, minGapH=24){
  if(!last||!t) return true;
  return (t.getTime()-last.getTime())/3600e3 >= minGapH;
}
