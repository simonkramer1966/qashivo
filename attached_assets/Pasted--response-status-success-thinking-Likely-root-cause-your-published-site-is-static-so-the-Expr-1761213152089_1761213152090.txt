Example Express route sketch:

app.get('/media/:filename', async (req, res) => {
  const filename = req.params.filename;
  const range = req.headers.range;

  // If your SDK supports ranged reads, use it. Example:
  // const { body, size, mime } = await objectStorageClient.get(`public/${filename}`, { range });
  // Fallback: fetch full bytes and slice in memory (works for ~30MB, not ideal).

  const result = await objectStorageClient.downloadAsBytes(`public/${filename}`);
  if (result.error) return res.sendStatus(404);
  const buf = result.value[0];
  const size = buf.length;

  res.setHeader('Content-Type', 'video/mp4');
  res.setHeader('Accept-Ranges', 'bytes');
  res.setHeader('Cache-Control', 'public, max-age=86400');

  if (!range) {
    res.setHeader('Content-Length', size);
    return res.status(200).send(buf);
  }

  const m = range.match(/bytes=(\d+)-(\d*)/);
  const start = parseInt(m[1], 10);
  const end = m[2] ? Math.min(parseInt(m[2], 10), size - 1) : Math.min(start + 1024 * 1024 - 1, size - 1); // 1MB chunks
  const chunk = buf.subarray(start, end + 1);

  res.status(206);
  res.setHeader('Content-Range', `bytes ${start}-${end}/${size}`);
  res.setHeader('Content-Length', chunk.length);
  res.send(chunk);
});
