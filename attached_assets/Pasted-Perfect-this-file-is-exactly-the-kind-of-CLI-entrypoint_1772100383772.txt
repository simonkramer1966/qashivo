Perfect — this file is exactly the kind of “CLI entrypoint that should never auto-run when imported”.

Here’s the safe ESM version that:

✅ Does not auto-execute on import

✅ Removes process.exit() from inside main()

✅ Keeps clean CLI behaviour only when run as the entrypoint

✅ Still closes the MCP server on SIGINT

Replace server/mcp/index.ts with this:

#!/usr/bin/env node

import { McpServer } from "@modelcontextprotocol/sdk/server/mcp.js";
import { StdioServerTransport } from "@modelcontextprotocol/sdk/server/stdio.js";
import { registerCallTools } from "./tools/call.js";
import { registerEmailTools } from "./tools/email.js";
import { registerCallOutcomeTools } from "./tools/callOutcomes.js";
import { createRetellClient } from "./client.js";

import { fileURLToPath } from "url";
import path from "path";

function createMcpServer() {
  const retellApiKey = process.env.RETELL_API_KEY;
  if (!retellApiKey) {
    throw new Error("RETELL_API_KEY environment variable is required");
  }

  const retellClient = createRetellClient(retellApiKey);

  const mcpServer = new McpServer({
    name: "Nexus AR Retell MCP",
    version: "1.0.0",
    capabilities: [],
  });

  registerCallTools(mcpServer, retellClient);
  registerEmailTools(mcpServer);
  registerCallOutcomeTools(mcpServer);

  return mcpServer;
}

export async function main() {
  const mcpServer = createMcpServer();
  const transport = new StdioServerTransport();
  await mcpServer.connect(transport);

  // IMPORTANT: no process.exit() inside main()
  process.on("SIGINT", async () => {
    try {
      await mcpServer.close();
      // return only — let the CLI wrapper decide whether to exit
    } catch (err) {
      console.error("Error shutting down MCP server:", err);
      // return only
    }
  });
}

// ESM-safe "run only if executed directly" guard
const isMain =
  process.argv[1] !== undefined &&
  path.resolve(process.argv[1]) === path.resolve(fileURLToPath(import.meta.url));

if (isMain) {
  main()
    .then(() => {
      // Only exit in true CLI execution
      process.exit(0);
    })
    .catch((err) => {
      console.error("Error starting MCP server:", err);
      process.exit(1);
    });
}
Why this works

main() is now safe to import (no exits, no auto-run).

The if (isMain) guard ensures it only runs when invoked directly by Node.

Exits are confined to the CLI wrapper block, not the shared logic.

Quick acceptance checks

Grep for bare main() calls

Should only appear inside the if (isMain) block.

Grep for process.exit

Should only be in the bottom guarded block (CLI-only), not inside main().
