2) API endpoints (thin, idempotent)

POST /api/actions/:id/approve → uses override if present, transitions to approved, enqueues send.

POST /api/actions/:id/edit → upserts override and returns updated preview.

POST /api/actions/:id/snooze → body: { until: ISO, reason? } → status snoozed.

POST /api/actions/:id/escalate → body: { queue: "Legal"|"Hardship"|..., note? } → status escalated.

POST /api/actions/:id/assign → { userId }.

POST /api/actions/:id/feedback → { value: "up"|"down", note? }.

Safety: reject approve if within tenant quiet hours; log and auto-move send time to first safe slot.

This replaces the “Call” link with: primary CTA, why? popover, inline triage, thumbs, and safe batching hook.

// components/ActionCentre/NextActionCell.tsx
import { useState } from "react";

type Reason = { label: string }; // plain english lines
type ActionRow = {
  id: string;
  priority: number; // 0..100
  recommended: { channel: "email"|"sms"|"voice"; sendAt: string; };
  reasons: Reason[]; // 1-3 bullets, plain english
  exceptionTags: string[];
  slaAgeMinutes: number; // from firstSeenAt
};

export default function NextActionCell({
  row,
  onApprove,
  onEdit,
  onSnooze,
  onEscalate,
  onFeedback
}: {
  row: ActionRow;
  onApprove: (id: string) => Promise<void>;
  onEdit: (id: string) => void;         // open side drawer
  onSnooze: (id: string, until: string) => Promise<void>;
  onEscalate: (id: string) => void;     // open escalate modal
  onFeedback: (id: string, val: "up"|"down") => Promise<void>;
}) {
  const [open, setOpen] = useState(false);
  const [busy, setBusy] = useState<null | "approve" | "snooze">(null);

  const label = labelFromRec(row.recommended); // e.g., "Call today after 10:00" or "Send SMS tomorrow 10:00"
  const isHigh = row.priority >= 70;

  return (
    <div className="flex items-start gap-2">
      <button
        className={`px-3 py-1.5 rounded-2xl text-sm font-medium shadow-sm
          ${isHigh ? "bg-emerald-600 text-white" : "bg-slate-800 text-white"}`}
        disabled={busy!==null}
        onClick={async ()=>{
          setBusy("approve");
          await onApprove(row.id);
          setBusy(null);
        }}
        aria-label={`Approve next action: ${label}`}
      >
        {busy==="approve" ? "Scheduling…" : label}
      </button>

      <button
        className="text-sky-700 underline text-xs mt-2"
        onClick={()=>setOpen(v=>!v)}
        aria-expanded={open}
        aria-controls={`why-${row.id}`}
      >
        why?
      </button>

      {/* Inline triage */}
      <div className="ml-auto flex gap-2">
        <button className="text-xs px-2 py-1 rounded bg-slate-200" onClick={()=>onEdit(row.id)}>Edit</button>
        <button
          className="text-xs px-2 py-1 rounded bg-slate-200"
          disabled={busy!==null}
          onClick={async()=>{
            setBusy("snooze");
            const until = nextBusinessMorningISO(); // implement per tenant tz
            await onSnooze(row.id, until);
            setBusy(null);
          }}
        >Snooze</button>
        <button className="text-xs px-2 py-1 rounded bg-rose-100 text-rose-700" onClick={()=>onEscalate(row.id)}>Escalate</button>
      </div>

      {/* Thumbs feedback */}
      <div className="flex items-center gap-1 pl-2">
        <button className="text-xs" aria-label="Good recommendation" onClick={()=>onFeedback(row.id,"up")}>👍</button>
        <button className="text-xs" aria-label="Poor recommendation" onClick={()=>onFeedback(row.id,"down")}>👎</button>
      </div>

      {/* Explainability popover (simple) */}
      {open && (
        <div id={`why-${row.id}`} className="absolute z-10 mt-6 w-64 rounded-xl border bg-white p-3 shadow-lg">
          <ul className="text-xs text-slate-700 space-y-1">
            {row.reasons.slice(0,3).map((r,i)=><li key={i}>• {r.label}</li>)}
          </ul>
          <div className="mt-2 flex items-center justify-between">
            <div className="text-[10px] text-slate-500">
              Priority: {row.priority >= 70 ? "High" : row.priority >= 50 ? "Medium" : "Low"}
            </div>
            {row.slaAgeMinutes>1440 && <span className="text-[10px] text-amber-600">SLA &gt; 24h</span>}
          </div>
        </div>
      )}
    </div>
  );
}

function labelFromRec(r:{channel:string; sendAt:string}) {
  const dt = new Date(r.sendAt);
  const when = humanizeTime(dt); // “today after 10:00”, “tomorrow 09:30”
  const verb = r.channel==="voice" ? "Call" : r.channel==="sms" ? "Send SMS" : "Send email";
  return `${verb} ${when}`;
}
function humanizeTime(dt:Date){ /* stub: format relative day + HH:mm */ return `tomorrow ${pad(dt.getHours())}:${pad(dt.getMinutes())}` }
function nextBusinessMorningISO(){ /* stub for snooze */ return new Date(Date.now()+24*3600e3).toISOString(); }
const pad=(n:number)=>String(n).padStart(2,"0");
