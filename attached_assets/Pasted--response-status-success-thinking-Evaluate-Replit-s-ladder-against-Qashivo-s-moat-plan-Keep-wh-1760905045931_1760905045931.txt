// src/policy/guards.ts
export function guardCheck(ctx: { now: Date; consent: Record<string, boolean>; quietHours: [string,string]; paused: boolean }) {
  if (ctx.paused) return { allowed: false, reason: "paused" };
  const [start,end] = ctx.quietHours;
  const t = ctx.now.toTimeString().slice(0,5);
  if ((start < end && t >= start && t < end) || (start > end && (t >= start || t < end))) {
    return { allowed: false, reason: "quiet_hours" };
  }
  return { allowed: true };
}

// src/policy/score.ts
export type ScoreInputs = { daysOverdue: number; amount: number; daysSinceLastContact: number; reliability: number; hasActiveDispute: boolean };
export function computeScore(i: ScoreInputs, w = { d:1.5, a:0.3/1000, l:0.5, r:-1.0, disp:-10 }) {
  let s = i.daysOverdue * w.d + i.amount * w.a + i.daysSinceLastContact * w.l + i.reliability * w.r;
  if (i.hasActiveDispute) s += w.disp;
  return Math.max(0, s);
}

export function schedule(ctx: any) {

  if (!guard.allowed) return emitDecision({ invoiceId: ctx.invoiceId, score: 0, factors: [guard.reason], policyVersion: "v1", ctx });
  const score = computeScore(ctx.inputs);
